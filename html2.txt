<script>
    // Numerical Integration using Simpson's 1/3 Rule
    function integrate(f, a, b, n = 1000) {
        if (n % 2 !== 0) n++; // Ensure n is even
        const h = (b - a) / n;
        let sum = f(a) + f(b);
        
        for (let i = 1; i < n; i += 2) {
            sum += 4 * f(a + i * h);
        }
        for (let i = 2; i < n - 1; i += 2) {
            sum += 2 * f(a + i * h);
        }
        
        return (sum * h) / 3;
    }

    function calculateAndPlot() {
        const funcStr = document.getElementById('function').value;
        const a = parseFloat(document.getElementById('lowerBound').value);
        const b = parseFloat(document.getElementById('upperBound').value);
        const resultBox = document.getElementById('resultBox');

        resultBox.style.display = 'block';
        resultBox.className = 'result';

        if (isNaN(a) || isNaN(b) || a >= b) {
            resultBox.textContent = 'Error: Ensure bounds are numbers and a < b.';
            resultBox.classList.add('error');
            return;
        }

        try {
            // Compile the function using math.js
            const node = math.parse(funcStr);
            const code = node.compile();
            const f = (x) => code.evaluate({ x: x });

            // Calculate Integral Area
            const area = integrate(f, a, b);
            resultBox.innerHTML = `Area &approx; ${area.toFixed(4)} &nbsp;&nbsp; <span style="font-weight:normal; font-size:14px">(using numerical integration)</span>`;

            // Prepare data for Plotly
            const margin = (b - a) * 0.5; // Add visual margin to the sides
            const xMin = a - margin;
            const xMax = b + margin;
            const step = (xMax - xMin) / 300;

            const xValues = [], yValues = [];
            const xFill = [], yFill = [];

            for (let x = xMin; x <= xMax; x += step) {
                const y = f(x);
                // Protect against infinities/NaNs disrupting the plot completely
                if (isFinite(y)) {
                    xValues.push(x);
                    yValues.push(y);
                    
                    if (x >= a && x <= b) {
                        xFill.push(x);
                        yFill.push(y);
                    }
                }
            }

            // Also ensure exact bounds are pushed into the fill array
            xFill.unshift(a); yFill.unshift(f(a));
            xFill.push(b); yFill.push(f(b));

            // Create plot traces
            const traceLine = {
                x: xValues,
                y: yValues,
                type: 'scatter',
                mode: 'lines',
                name: 'f(x)',
                line: { color: '#2980b9', width: 2 }
            };

            const traceFill = {
                x: xFill,
                y: yFill,
                fill: 'tozeroy', // Fills down to the x-axis
                type: 'scatter',
                mode: 'none',
                name: `Area [${a}, ${b}]`,
                fillcolor: 'rgba(52, 152, 219, 0.4)'
            };

            const layout = {
                title: `Graph of ${funcStr}`,
                margin: { t: 40, r: 20, l: 40, b: 40 },
                xaxis: { zeroline: true, zerolinecolor: '#999' },
                yaxis: { zeroline: true, zerolinecolor: '#999' },
                showlegend: false
            };

            Plotly.newPlot('plot', [traceFill, traceLine], layout, {staticPlot: false});

        } catch (error) {
            resultBox.textContent = 'Error: Invalid function. Use proper math syntax (e.g., sin(x), x^2).';
            resultBox.classList.add('error');
            Plotly.purge('plot'); // Clear plot on error
        }
    }

    // Run once on load to show initial graph
    window.onload = calculateAndPlot;
</script>
